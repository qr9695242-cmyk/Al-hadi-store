// جاؤ Firebase Console > Firestore > Rules اور یہ rules پیسٹ کریں:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Admin products collection
    match /products/{productId} {
      allow read: if true;
      allow write: if false; // صرف backend/admin SDK سے write ہو سکے
    }
    
    // User liked products
    match /users/{userId} {
      allow read: if request.auth.uid == userId;
      allow write: if request.auth.uid == userId;
    }
  }
}
